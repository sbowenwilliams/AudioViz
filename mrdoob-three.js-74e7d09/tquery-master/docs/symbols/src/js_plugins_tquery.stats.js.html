  2  * @fileOverview Plugins for tQuery and Stats.js

  4 
  5 (function(){
  6 
  7 /**
  8  * 
  9 */
 10 var myStats	= function(loop)
 11 {
 12 	// add Stats.js - https://github.com/mrdoob/stats.js
 13 	this._stats	= new Stats();
 14 	this._stats.domElement.style.position	= 'absolute';
 15 	this._stats.domElement.style.bottom	= '0px';
 16 	document.body.appendChild( this._stats.domElement );
 17 
 18 	this._loop	= loop	|| tQuery.world.loop();
 19 
 20 	this._$onRender	= this._onRender.bind(this);
 21 	this._loop.hookPostRender(this._$onRender);
 22 }
 23 
 24 myStats.prototype.destroy	= function(){
 25 	this._loop.unhookPostRender(this._$onRender);	
 26 	document.body.removeChild(this._stats.domElement );
 27 }
 28 
 29 myStats.prototype._onRender	= function(){
 30 	this._stats.update();
 31 };
 32 
 33 // register the plugins
 34 tQuery.register('Stats', myStats);
 35 tQuery.register('createStats', function(loop){ return new tQuery.Stats(loop); });
 36 
 37 })();	// closure function end
