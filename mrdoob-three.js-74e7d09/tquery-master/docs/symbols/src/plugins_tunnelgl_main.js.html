 86 var vertexTransform	= function(o, v, time){
 87 	var z		= (playerZ + v.z);
 88 	var angle	= z * 0.8;
 89 	var rangeX	= 0.15;
 90 	var rangeY	= 0.1;
 91 	
 92 	//rangeX	= rangeY	= 0;
 93 	
 94 	v.x	= o.x + Math.cos(angle*0.5)*rangeX;
 95 	v.y	= o.y + Math.sin(angle*0.5)*rangeY;
 96 };
