 10 var DragPanControls	= function(loop)
 11 {
 12 	this._loop	= loop	|| tQuery.world.loop();
 13 
 14 	this._controls	= new THREEx.DragPanControls(tQuery.world.camera());
 15 	this._$onRender	= this._onRender.bind(this);
 16 	this._loop.hookPreRender(this._$onRender);
 17 };
 18 
 19 DragPanControls.prototype.destroy	= function(){
 20 	this._loop.unhookPreRender(this._$onRender);	
 21 };
 22 
 23 DragPanControls.prototype._onRender	= function(){
 24 	this._controls.update();
 25 };
 26 
 27 
 28 // register the plugins
 29 tQuery.register('DragPanControls', DragPanControls);
 30 tQuery.register('createDragPanControls', function(loop){ return new tQuery.DragPanControls(loop); });
 31 
 32 })();	// closure function end
