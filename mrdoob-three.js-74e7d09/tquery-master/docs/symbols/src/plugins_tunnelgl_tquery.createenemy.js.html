1 tQuery.register('createEnemy', function(opts){
  2 	// handle parameters
  3 	opts	= tQuery.extend(opts, {		
  4 		id		: undefined,
  5 		loop		: tQuery.world.loop(),
  6 		getSpeedZ	: function(){
  7 			return Math.random()*0.05+0.1
  8 		}
  9 	});
 10 	console.assert(opts.id, "invalid parameter");
 11 
 12 	var radius	= 0.45;
 13 	var speedZ	= opts.getSpeedZ();
 14 	
 15 	var material	= new THREE.MeshNormalMaterial();
 16 	var material	= new THREE.MeshLambertMaterial({
 17 		ambient	: 0x444444,
 18 		color	: 0x88AAFF,
 19 		map	: waterTexture
 20 	});
 21 	var geometry	= new THREE.SphereGeometry(0.5, 32, 16)
 22 	var object	= tQuery(new THREE.Mesh(geometry, material));
 23 	object.id(opts.id).addTo(tQuery.world);
 24 	
 25 	
 26 	var angle	= Math.random() * Math.PI * 2;
 27 
 28 	object.geometry()
 29 		.zoom(0.2)
 30 		.rotateZ(angle + Math.PI/2);
 31 
 32 	var positionX	= Math.cos(angle) * radius;

 35 	object.position(positionX, positionY, positionZ)
 36 
 37 	opts.loop.hook(function(deltaTime, present){
 38 		var tMesh	= object.get(0)
 39 
 40 		tMesh.position.z	+= speedZ;		
 41 		if( tMesh.position.z > 4 )	onDeath();
 42 
 43 		var origin	= {
 44 			x	: Math.cos(angle) * radius,
 45 			y	: Math.sin(angle) * radius,
 46 			z	:  0
 47 		};
 48 		vertexTransform(origin, object.get(0).position, present)
 49 	});
 50 	
 51 
 52 	function onDeath(){
 53 		speedZ	= opts.getSpeedZ();
 54 		
 55 		var angle	= Math.random() * Math.PI * 2;	
 56 		var positionX	= Math.cos(angle) * radius;
 57 		var positionY	= Math.sin(angle) * radius;
 58 		var positionZ	= -8;
 59 		object.position(positionX, positionY, positionZ)		
 60 	}
 61 });
