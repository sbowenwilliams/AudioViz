1 /**
  2  * tquery.js plugin to handle webaudio
  3 */
  4 define(['examples/webaudio/threex.webaudio'], function(){
  5 	var instance	= null;
  6 	tQuery.World.register('webaudio', function(){		
  7 		if( !instance )	instance = new tQuery.WebAudio();
  8 		return instance;
  9 	});
 10 
 11 	tQuery.WebAudio	= function(){
 12 		this._webaudio	= new THREEx.WebAudio();
 13 	};
 14 	
 15 	tQuery.WebAudio.prototype.get	= function(){
 16 		return this._webaudio;
 17 	}
 18 
 19 	// Proxy raw functions
 20 	tQuery.WebAudio._rawFn2proxy	= ['context', 'volume'];
 21 	tQuery.WebAudio._rawFn2proxy.forEach(function(fnName){
 22 		tQuery.WebAudio.prototype[fnName]	= function(){
 23 			return this._webaudio[fnName].apply(this._webaudio, arguments)
 24 		};
 25 	});
 26 
 27 	// Proxy chained functions
 28 	tQuery.WebAudio._chainedFn2proxy	= [];
 29 	tQuery.WebAudio._chainedFn2proxy.forEach(function(fnName){
 30 		tQuery.WebAudio.prototype[fnName]	= function(){
 31 			this._webaudio[fnName].apply(this._webaudio, arguments)
 32 			return this;	// for chained API 
 33 		};
 34 	});
 35 });
