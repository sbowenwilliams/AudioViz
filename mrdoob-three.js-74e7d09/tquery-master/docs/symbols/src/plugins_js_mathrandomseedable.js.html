 69 var MersenneTwister = function(seed) {
 70   if (seed == undefined) {
 71     seed = new Date().getTime();
 72   } 
 73   /* Period parameters */  
 74   this.N = 624;
 75   this.M = 397;
 76   this.MATRIX_A = 0x9908b0df;   /* constant vector a */
 77   this.UPPER_MASK = 0x80000000; /* most significant w-r bits */
 78   this.LOWER_MASK = 0x7fffffff; /* least significant r bits */
 79  
 80   this.mt = new Array(this.N); /* the array for the state vector */
 81   this.mti=this.N+1; /* mti==N+1 means mt[N] is not initialized */
 82 
 83   this.init_genrand(seed);
 84 }  
 85  
 86 /* initializes mt[N] with a seed */
 87 MersenneTwister.prototype.init_genrand = function(s) {
 88   this.mt[0] = s >>> 0;
 89   for (this.mti=1; this.mti<this.N; this.mti++) {
 90       var s = this.mt[this.mti-1] ^ (this.mt[this.mti-1] >>> 30);
 91    this.mt[this.mti] = (((((s & 0xffff0000) >>> 16) * 1812433253) << 16) + (s & 0x0000ffff) * 1812433253)
 92   + this.mti;
 93       /* See Knuth TAOCP Vol2. 3rd Ed. P.106 for multiplier. */
 94       /* In the previous versions, MSBs of the seed affect   */
 95       /* only MSBs of the array mt[].                        */
 96       /* 2002/01/09 modified by Makoto Matsumoto             */
 97       this.mt[this.mti] >>>= 0;
 98       /* for >32 bit machines */
 99   }
100 }

102 /* initialize by an array with array-length */
103 /* init_key is the array for initializing keys */
104 /* key_length is its length */
105 /* slight change for C++, 2004/2/26 */
106 MersenneTwister.prototype.init_by_array = function(init_key, key_length) {
107   var i, j, k;
108   this.init_genrand(19650218);
109   i=1; j=0;
110   k = (this.N>key_length ? this.N : key_length);
111   for (; k; k--) {
112     var s = this.mt[i-1] ^ (this.mt[i-1] >>> 30)
113     this.mt[i] = (this.mt[i] ^ (((((s & 0xffff0000) >>> 16) * 1664525) << 16) + ((s & 0x0000ffff) * 1664525)))
114       + init_key[j] + j; /* non linear */
115     this.mt[i] >>>= 0; /* for WORDSIZE > 32 machines */
116     i++; j++;
117     if (i>=this.N) { this.mt[0] = this.mt[this.N-1]; i=1; }
118     if (j>=key_length) j=0;
119   }
120   for (k=this.N-1; k; k--) {
121     var s = this.mt[i-1] ^ (this.mt[i-1] >>> 30);
122     this.mt[i] = (this.mt[i] ^ (((((s & 0xffff0000) >>> 16) * 1566083941) << 16) + (s & 0x0000ffff) * 1566083941))
123       - i; /* non linear */
124     this.mt[i] >>>= 0; /* for WORDSIZE > 32 machines */
125     i++;
126     if (i>=this.N) { this.mt[0] = this.mt[this.N-1]; i=1; }
127   }
128 
129   this.mt[0] = 0x80000000; /* MSB is 1; assuring non-zero initial array */ 
130 }

132 /* generates a random number on [0,0xffffffff]-interval */
133 MersenneTwister.prototype.genrand_int32 = function() {
134   var y;
135   var mag01 = new Array(0x0, this.MATRIX_A);
136   /* mag01[x] = x * MATRIX_A  for x=0,1 */
137 
138   if (this.mti >= this.N) { /* generate N words at one time */
139     var kk;
140 
141     if (this.mti == this.N+1)   /* if init_genrand() has not been called, */
142       this.init_genrand(5489); /* a default initial seed is used */
143 
144     for (kk=0;kk<this.N-this.M;kk++) {
145       y = (this.mt[kk]&this.UPPER_MASK)|(this.mt[kk+1]&this.LOWER_MASK);
146       this.mt[kk] = this.mt[kk+this.M] ^ (y >>> 1) ^ mag01[y & 0x1];
147     }
148     for (;kk<this.N-1;kk++) {
149       y = (this.mt[kk]&this.UPPER_MASK)|(this.mt[kk+1]&this.LOWER_MASK);
150       this.mt[kk] = this.mt[kk+(this.M-this.N)] ^ (y >>> 1) ^ mag01[y & 0x1];
151     }
152     y = (this.mt[this.N-1]&this.UPPER_MASK)|(this.mt[0]&this.LOWER_MASK);
153     this.mt[this.N-1] = this.mt[this.M-1] ^ (y >>> 1) ^ mag01[y & 0x1];
154 
155     this.mti = 0;
156   }
157 
158   y = this.mt[this.mti++];
159 
160   /* Tempering */
161   y ^= (y >>> 11);
162   y ^= (y << 7) & 0x9d2c5680;
163   y ^= (y << 15) & 0xefc60000;
164   y ^= (y >>> 18);
165 
166   return y >>> 0;
167 }

169 /* generates a random number on [0,0x7fffffff]-interval */
170 MersenneTwister.prototype.genrand_int31 = function() {
171   return (this.genrand_int32()>>>1);
172 }
173  
174 /* generates a random number on [0,1]-real-interval */
175 MersenneTwister.prototype.genrand_real1 = function() {
176   return this.genrand_int32()*(1.0/4294967295.0); 
177   /* divided by 2^32-1 */ 
178 }
179 
180 /* generates a random number on [0,1)-real-interval */
181 MersenneTwister.prototype.random = function() {
182   return this.genrand_int32()*(1.0/4294967296.0); 
183   /* divided by 2^32 */
184 }
185  
186 /* generates a random number on (0,1)-real-interval */
187 MersenneTwister.prototype.genrand_real3 = function() {
188   return (this.genrand_int32() + 0.5)*(1.0/4294967296.0); 
189   /* divided by 2^32 */
190 }

192 /* generates a random number on [0,1) with 53-bit resolution*/
193 MersenneTwister.prototype.genrand_res53 = function() { 
194   var a=this.genrand_int32()>>>5, b=this.genrand_int32()>>>6; 
195   return(a*67108864.0+b)*(1.0/9007199254740992.0); 
196 }
