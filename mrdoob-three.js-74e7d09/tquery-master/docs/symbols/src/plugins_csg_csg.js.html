1 // Constructive Solid Geometry (CSG) is a modeling technique that uses Boolean
  2 // operations like union and intersection to combine 3D solids. This library
  3 // implements CSG operations on meshes elegantly and concisely using BSP trees,
  4 // and is meant to serve as an easily understandable implementation of the
  5 // algorithm. All edge cases involving overlapping coplanar polygons in both
  6 // solids are correctly handled.
  7 // 
  8 // Example usage:
  9 // 
 10 //     var cube = CSG.cube();
 11 //     var sphere = CSG.sphere({ radius: 1.3 });
 12 //     var polygons = cube.subtract(sphere).toPolygons();
 13 // 
 14 // ## Implementation Details
 15 // 
 16 // All CSG operations are implemented in terms of two functions, `clipTo()` and
 17 // `invert()`, which remove parts of a BSP tree inside another BSP tree and swap
 18 // solid and empty space, respectively. To find the union of `a` and `b`, we
 19 // want to remove everything in `a` inside `b` and everything in `b` inside `a`,
 20 // then combine polygons from `a` and `b` into one solid:
 21 // 
 22 //     a.root.clipTo(b.root);
 23 //     b.root.clipTo(a.root);
 24 //     a.root.build(b.root.allPolygons());
 25 // 
 26 // The only tricky part is handling overlapping coplanar polygons in both trees.
 27 // The code above keeps both copies, but we need to keep them in one tree and
 28 // remove them in the other tree. To remove them from `b` we can clip the
 29 // inverse of `b` against `a`. The code for union now looks like this:
 30 // 
 31 //     a.root.clipTo(b.root);
 32 //     b.root.clipTo(a.root);
 33 //     b.root.invert();
 34 //     b.root.clipTo(a.root);
 35 //     b.root.invert();
 36 //     a.root.build(b.root.allPolygons());
 37 // 
 38 // Subtraction and intersection naturally follow from set operations. If
 39 // union is `A | B`, subtraction is `A - B = ~(~A | B)` and intersection is
 40 // `A & B = ~(~A | ~B)` where `~` is the complement operator.
 41 //

 46 // # class CSG
 47 
 48 // Holds a binary space partition tree representing a 3D solid. Two solids can
 49 // be combined using the `union()`, `subtract()`, and `intersect()` methods.
 50 
 51 var CSG = function() {
 52   this.root = new CSG.Node();
 53 };
 54 
 55 // Construct a CSG solid from a list of `CSG.Polygon` instances.
 56 CSG.fromPolygons = function(polygons) {
 57   var bsp = new CSG();
 58   bsp.root.build(polygons);
 59   return bsp;
 60 };
 61 
 62 CSG.prototype = {
 63   clone: function() {
 64     var bsp = new CSG();
 65     bsp.root = this.root.clone();
 66     return bsp;
 67   },
 68 
 69   toPolygons: function() {
 70     return this.root.allPolygons();
 71   },
 72 
 73   // Return a new CSG solid representing space in either this solid or in the
 74   // solid `bsp`. Neither this solid nor the solid `bsp` are modified.
 75   // 
 76   //     A.union(B)
 77   // 
 78   //     +-------+            +-------+
 79   //     |       |            |       |
 80   //     |   A   |            |       |
 81   //     |    +--+----+   =   |       +----+
 82   //     +----+--+    |       +----+       |
 83   //          |   B   |            |       |
 84   //          |       |            |       |
 85   //          +-------+            +-------+
 86   // 
 87   union: function(bsp) {
 88     var a = this.clone(), b = bsp.clone();
 89     a.root.clipTo(b.root);
 90     b.root.clipTo(a.root);
 91     b.root.invert();
 92     b.root.clipTo(a.root);
 93     b.root.invert();
 94     a.root.build(b.root.allPolygons());
 95     return a;
 96   },
 97 
 98   // Return a new CSG solid representing space in this solid but not in the
 99   // solid `bsp`. Neither this solid nor the solid `bsp` are modified.
100   // 
101   //     A.subtract(B)
102   // 
103   //     +-------+            +-------+
104   //     |       |            |       |
105   //     |   A   |            |       |
106   //     |    +--+----+   =   |    +--+
107   //     +----+--+    |       +----+
108   //          |   B   |
109   //          |       |
110   //          +-------+
111   // 
112   subtract: function(bsp) {
113     var a = this.clone(), b = bsp.clone();
114     a.root.invert();
115     a.root.clipTo(b.root);
116     b.root.clipTo(a.root);
117     b.root.invert();
118     b.root.clipTo(a.root);
119     b.root.invert();
120     a.root.build(b.root.allPolygons());
121     a.root.invert();
122     return a;
123   },
124 
125   // Return a new CSG solid representing space both this solid and in the
126   // solid `bsp`. Neither this solid nor the solid `bsp` are modified.
127   // 
128   //     A.intersect(B)
129   // 
130   //     +-------+
131   //     |       |
132   //     |   A   |
133   //     |    +--+----+   =   +--+
134   //     +----+--+    |       +--+
135   //          |   B   |
136   //          |       |
137   //          +-------+
138   // 
139   intersect: function(bsp) {
140     var a = this.clone(), b = bsp.clone();
141     a.root.invert();
142     b.root.clipTo(a.root);
143     b.root.invert();
144     a.root.clipTo(b.root);
145     b.root.clipTo(a.root);
146     a.root.build(b.root.allPolygons());
147     a.root.invert();
148     return a;
149   },
150 
151   // Return a new CSG solid with solid and empty space switched. This solid is
152   // not modified.
153   inverse: function() {
154     var bsp = this.clone();
155     bsp.root.invert();
156     return bsp;
157   }
158 };

160 // Construct an axis-aligned solid cube. Optional parameters are `center` and
161 // `radius`, which default to `[0, 0, 0]` and `1`.
162 // 
163 // Example code:
164 // 
165 //     var cube = CSG.cube({
166 //       center: [0, 0, 0],
167 //       radius: 1
168 //     });
169 CSG.cube = function(options) {
170   options = options || {};
171   var c = new CSG.Vector(options.center || [0, 0, 0]);
172   var r = options.radius || 1;
173   return CSG.fromPolygons([
174     [[0, 4, 6, 2], [-1, 0, 0]],
175     [[1, 3, 7, 5], [+1, 0, 0]],
176     [[0, 1, 5, 4], [0, -1, 0]],
177     [[2, 6, 7, 3], [0, +1, 0]],
178     [[0, 2, 3, 1], [0, 0, -1]],
179     [[4, 5, 7, 6], [0, 0, +1]]
180   ].map(function(info) {
181     return new CSG.Polygon(info[0].map(function(i) {
182       var pos = new CSG.Vector(
183         c.x + r * (2 * !!(i & 1) - 1),
184         c.y + r * (2 * !!(i & 2) - 1),
185         c.z + r * (2 * !!(i & 4) - 1)
186       );
187       return new CSG.Vertex(pos, new CSG.Vector(info[1]));
188     }));
189   }));
190 };

192 // Construct a solid sphere. Optional parameters are `center`, `radius`,
193 // `slices`, and `stacks`, which default to `[0, 0, 0]`, `1`, `16`, and `8`.
194 // The `slices` and `stacks` parameters control the tessellation along the
195 // longitude and latitude directions.
196 // 
197 // Example usage:
198 // 
199 //     var sphere = CSG.sphere({
200 //       center: [0, 0, 0],
201 //       radius: 1,
202 //       slices: 16,
203 //       stacks: 8
204 //     });
205 CSG.sphere = function(options) {
206   options = options || {};
207   var c = new CSG.Vector(options.center || [0, 0, 0]);
208   var r = options.radius || 1;
209   var slices = options.slices || 16;
210   var stacks = options.stacks || 8;
211   var polygons = [], vertices;
212   function vertex(theta, phi) {
213     theta *= Math.PI * 2;
214     phi *= Math.PI;
215     var dir = new CSG.Vector(
216       Math.cos(theta) * Math.sin(phi),
217       Math.cos(phi),
218       Math.sin(theta) * Math.sin(phi)
219     );
220     vertices.push(new CSG.Vertex(c.plus(dir.times(r)), dir));
221   }
222   for (var i = 0; i < slices; i++) {
223     for (var j = 0; j < stacks; j++) {
224       vertices = [];
225       vertex(i / slices, j / stacks);
226       if (j > 0) vertex((i + 1) / slices, j / stacks);
227       if (j < stacks - 1) vertex((i + 1) / slices, (j + 1) / stacks);
228       vertex(i / slices, (j + 1) / stacks);
229       polygons.push(new CSG.Polygon(vertices));
230     }
231   }
232   return CSG.fromPolygons(polygons);
233 };

235 // Construct a solid cylinder. Optional parameters are `start`, `end`,
236 // `radius`, and `slices`, which default to `[0, -1, 0]`, `[0, 1, 0]`, `1`, and
237 // `16`. The `slices` parameter controls the tessellation.
238 // 
239 // Example usage:
240 // 
241 //     var cylinder = CSG.cylinder({
242 //       start: [0, -1, 0],
243 //       end: [0, 1, 0],
244 //       radius: 1,
245 //       slices: 16
246 //     });
247 CSG.cylinder = function(options) {
248   options = options || {};
249   var s = new CSG.Vector(options.start || [0, -1, 0]);
250   var e = new CSG.Vector(options.end || [0, 1, 0]);
251   var ray = e.minus(s);
252   var r = options.radius || 1;
253   var slices = options.slices || 16;
254   var axisZ = ray.unit(), isY = (Math.abs(axisZ.y) > 0.5);
255   var axisX = new CSG.Vector(isY, !isY, 0).cross(axisZ).unit();
256   var axisY = axisX.cross(axisZ).unit();
257   var start = new CSG.Vertex(s, axisZ.negated());
258   var end = new CSG.Vertex(e, axisZ.unit());
259   var polygons = [];
260   function point(stack, slice, normalBlend) {
261     var angle = slice * Math.PI * 2;
262     var out = axisX.times(Math.cos(angle)).plus(axisY.times(Math.sin(angle)));
263     var pos = s.plus(ray.times(stack)).plus(out.times(r));
264     var normal = out.times(1 - Math.abs(normalBlend)).plus(axisZ.times(normalBlend));
265     return new CSG.Vertex(pos, normal);
266   }
267   for (var i = 0; i < slices; i++) {
268     var t0 = i / slices, t1 = (i + 1) / slices;
269     polygons.push(new CSG.Polygon([start, point(0, t0, -1), point(0, t1, -1)]));
270     polygons.push(new CSG.Polygon([point(0, t1, 0), point(0, t0, 0), point(1, t0, 0), point(1, t1, 0)]));
271     polygons.push(new CSG.Polygon([end, point(1, t1, 1), point(1, t0, 1)]));
272   }
273   return CSG.fromPolygons(polygons);
274 };

276 // # class Vector
277 
278 // Represents a 3D vector.
279 // 
280 // Example usage:
281 // 
282 //     new CSG.Vector(1, 2, 3);
283 //     new CSG.Vector([1, 2, 3]);
284 //     new CSG.Vector({ x: 1, y: 2, z: 3 });
285 
286 CSG.Vector = function(x, y, z) {
287   if (arguments.length == 3) {
288     this.x = x;
289     this.y = y;
290     this.z = z;
291   } else if ('x' in x) {
292     this.x = x.x;
293     this.y = x.y;
294     this.z = x.z;
295   } else {
296     this.x = x[0];
297     this.y = x[1];
298     this.z = x[2];
299   }
300 };
301 
302 CSG.Vector.prototype = {
303   clone: function() {
304     return new CSG.Vector(this.x, this.y, this.z);
305   },
306 
307   negated: function() {
308     return new CSG.Vector(-this.x, -this.y, -this.z);
309   },
310 
311   plus: function(a) {
312     return new CSG.Vector(this.x + a.x, this.y + a.y, this.z + a.z);
313   },
314 
315   minus: function(a) {
316     return new CSG.Vector(this.x - a.x, this.y - a.y, this.z - a.z);
317   },
318 
319   times: function(a) {
320     return new CSG.Vector(this.x * a, this.y * a, this.z * a);
321   },
322 
323   dividedBy: function(a) {
324     return new CSG.Vector(this.x / a, this.y / a, this.z / a);
325   },
326 
327   dot: function(a) {
328     return this.x * a.x + this.y * a.y + this.z * a.z;
329   },
330 
331   lerp: function(a, t) {
332     return this.plus(a.minus(this).times(t));
333   },
334 
335   length: function() {
336     return Math.sqrt(this.dot(this));
337   },
338 
339   unit: function() {
340     return this.dividedBy(this.length());
341   },
342 
343   cross: function(a) {
344     return new CSG.Vector(
345       this.y * a.z - this.z * a.y,
346       this.z * a.x - this.x * a.z,
347       this.x * a.y - this.y * a.x
348     );
349   }
350 };

352 // # class Vertex
353 
354 // Represents a vertex of a polygon. Use your own vertex class instead of this
355 // one to provide additional features like texture coordinates and vertex
356 // colors. Custom vertex classes need to provide a `pos` property and `clone()`,
357 // `flip()`, and `interpolate()` methods that behave analogous to the ones
358 // defined by `CSG.Vertex`. This class provides `normal` so convenience
359 // functions like `CSG.sphere()` can return a smooth vertex normal, but `normal`
360 // is not used anywhere else.
361 
362 CSG.Vertex = function(pos, normal) {
363   this.pos = new CSG.Vector(pos);
364   this.normal = new CSG.Vector(normal);
365 };
366 
367 CSG.Vertex.prototype = {
368   clone: function() {
369     return new CSG.Vertex(this.pos.clone(), this.normal.clone());
370   },
371 
372   // Invert all orientation-specific data (e.g. vertex normal). Called when the
373   // orientation of a polygon is flipped.
374   flip: function() {
375     this.normal = this.normal.negated();
376   },
377 
378   // Create a new vertex between this vertex and `other` by linearly
379   // interpolating all properties using a parameter of `t`. Subclasses should
380   // override this to interpolate additional properties.
381   interpolate: function(other, t) {
382     return new CSG.Vertex(
383       this.pos.lerp(other.pos, t),
384       this.normal.lerp(other.normal, t)
385     );
386   }
387 };

389 // # class Plane
390 
391 // Represents a plane in 3D space.
392 
393 CSG.Plane = function(normal, w) {
394   this.normal = normal;
395   this.w = w;
396 };
397 
398 // `CSG.Plane.EPSILON` is the tolerance used by `splitPolygon()` to decide if a
399 // point is on the plane.
400 CSG.Plane.EPSILON = 1e-5;
401 
402 CSG.Plane.fromPoints = function(a, b, c) {
403   var n = b.minus(a).cross(c.minus(a)).unit();
404   return new CSG.Plane(n, n.dot(a));
405 };
406 
407 CSG.Plane.prototype = {
408   clone: function() {
409     return new CSG.Plane(this.normal.clone(), this.w);
410   },
411 
412   flip: function() {
413     this.normal = this.normal.negated();
414     this.w = -this.w;
415   },
416 
417   // Split `polygon` by this plane if needed, then put the polygon or polygon
418   // fragments in the appropriate lists. Coplanar polygons go into either
419   // `coplanarFront` or `coplanarBack` depending on their orientation with
420   // respect to this plane. Polygons in front or in back of this plane go into
421   // either `front` or `back`.
422   splitPolygon: function(polygon, coplanarFront, coplanarBack, front, back) {
423     var COPLANAR = 0;
424     var FRONT = 1;
425     var BACK = 2;
426     var SPANNING = 3;
427 
428     // Classify each point as well as the entire polygon into one of the above
429     // four classes.
430     var polygonType = 0;
431     var types = [];
432     for (var i = 0; i < polygon.vertices.length; i++) {
433       var t = this.normal.dot(polygon.vertices[i].pos) - this.w;
434       var type = (t < -CSG.Plane.EPSILON) ? BACK : (t > CSG.Plane.EPSILON) ? FRONT : COPLANAR;
435       polygonType |= type;
436       types.push(type);
437     }
438 
439     // Put the polygon in the correct list, splitting it when necessary.
440     switch (polygonType) {
441       case COPLANAR:
442         (this.normal.dot(polygon.plane.normal) > 0 ? coplanarFront : coplanarBack).push(polygon);
443         break;
444       case FRONT:
445         front.push(polygon);
446         break;
447       case BACK:
448         back.push(polygon);
449         break;
450       case SPANNING:
451         var f = [], b = [];
452         for (var i = 0; i < polygon.vertices.length; i++) {
453           var j = (i + 1) % polygon.vertices.length;
454           var ti = types[i], tj = types[j];
455           var vi = polygon.vertices[i], vj = polygon.vertices[j];
456           if (ti != BACK) f.push(vi);
457           if (ti != FRONT) b.push(ti != BACK ? vi.clone() : vi);
458           if ((ti | tj) == SPANNING) {
459             var t = (this.w - this.normal.dot(vi.pos)) / this.normal.dot(vj.pos.minus(vi.pos));
460             var v = vi.interpolate(vj, t);
461             f.push(v);
462             b.push(v.clone());
463           }
464         }
465         if (f.length >= 3) front.push(new CSG.Polygon(f, polygon.shared));
466         if (b.length >= 3) back.push(new CSG.Polygon(b, polygon.shared));
467         break;
468     }
469   }
470 };

472 // # class Polygon
473 
474 // Represents a convex polygon. The vertices used to initialize a polygon must
475 // be coplanar and form a convex loop. They do not have to be `CSG.Vertex`
476 // instances but they must behave similarly (duck typing can be used for
477 // customization).
478 // 
479 // Each convex polygon has a `shared` property, which is shared between all
480 // polygons that are clones of each other or were split from the same polygon.
481 // This can be used to define per-polygon properties (such as surface color).
482 
483 CSG.Polygon = function(vertices, shared) {
484   this.vertices = vertices;
485   this.shared = shared;
486   this.plane = CSG.Plane.fromPoints(vertices[0].pos, vertices[1].pos, vertices[2].pos);
487 };
488 
489 CSG.Polygon.prototype = {
490   clone: function() {
491     var vertices = this.vertices.map(function(v) { return v.clone(); });
492     return new CSG.Polygon(vertices, this.shared);
493   },
494 
495   flip: function() {
496     this.vertices.reverse().map(function(v) { v.flip(); });
497     this.plane.flip();
498   }
499 };

501 // # class Node
502 
503 // Holds a node in a BSP tree. A BSP tree is built from a collection of polygons
504 // by picking a polygon to split along. That polygon (and all other coplanar
505 // polygons) are added directly to that node and the other polygons are added to
506 // the front and/or back subtrees. This is not a leafy BSP tree since there is
507 // no distinction between internal and leaf nodes.
508 
509 CSG.Node = function() {
510   this.plane = null;
511   this.front = null;
512   this.back = null;
513   this.polygons = [];
514 };
515 
516 CSG.Node.prototype = {
517   clone: function() {
518     var node = new CSG.Node();
519     node.plane = this.plane && this.plane.clone();
520     node.front = this.front && this.front.clone();
521     node.back = this.back && this.back.clone();
522     node.polygons = this.polygons.map(function(p) { return p.clone(); });
523     return node;
524   },
525 
526   // Convert solid space to empty space and empty space to solid space.
527   invert: function() {
528     for (var i = 0; i < this.polygons.length; i++) {
529       this.polygons[i].flip();
530     }
531     this.plane.flip();
532     if (this.front) this.front.invert();
533     if (this.back) this.back.invert();
534     var temp = this.front;
535     this.front = this.back;
536     this.back = temp;
537   },
538 
539   // Recursively remove all polygons in `polygons` that are inside this BSP
540   // tree.
541   clipPolygons: function(polygons) {
542     var front = [], back = [];
543     for (var i = 0; i < polygons.length; i++) {
544       this.plane.splitPolygon(polygons[i], front, back, front, back);
545     }
546     if (this.front) front = this.front.clipPolygons(front);
547     if (this.back) back = this.back.clipPolygons(back);
548     else back = [];
549     return front.concat(back);
550   },
551 
552   // Remove all polygons in this BSP tree that are inside the other BSP tree
553   // `bsp`.
554   clipTo: function(bsp) {
555     this.polygons = bsp.clipPolygons(this.polygons);
556     if (this.front) this.front.clipTo(bsp);
557     if (this.back) this.back.clipTo(bsp);
558   },
559 
560   // Return a list of all polygons in this BSP tree.
561   allPolygons: function() {
562     var polygons = this.polygons.slice();
563     if (this.front) polygons = polygons.concat(this.front.allPolygons());
564     if (this.back) polygons = polygons.concat(this.back.allPolygons());
565     return polygons;
566   },
567 
568   // Build a BSP tree out of `polygons`. When called on an existing tree, the
569   // new polygons are filtered down to the bottom of the tree and become new
570   // nodes there. Each set of polygons is partitioned using the first polygon
571   // (no heuristic is used to pick a good split).
572   build: function(polygons) {
573     if (!polygons.length) return;
574     if (!this.plane) this.plane = polygons[0].plane.clone();
575     var front = [], back = [];
576     for (var i = 0; i < polygons.length; i++) {
577       this.plane.splitPolygon(polygons[i], this.polygons, this.polygons, front, back);
578     }
579     if (front.length) {
580       if (!this.front) this.front = new CSG.Node();
581       this.front.build(front);
582     }
583     if (back.length) {
584       if (!this.back) this.back = new CSG.Node();
585       this.back.build(back);
586     }
587   }
588 };
