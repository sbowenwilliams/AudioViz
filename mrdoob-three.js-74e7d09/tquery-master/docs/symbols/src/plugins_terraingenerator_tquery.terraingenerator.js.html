  3 /**
  4  * Generate a light map
  5  * TODO make it more tunable
  6  * TODO make that usable standalone
  7 */
  8 function generateTexture( data, width, height ){
  9 
 10 	var sun		= new THREE.Vector3( 1, 0.5, 1 );
 11 	sun.normalize();
 12 
 13 	var canvas	= document.createElement( 'canvas' );
 14 	canvas.width	= width;
 15 	canvas.height	= height;
 16 
 17 	var context	= canvas.getContext( '2d' );
 18 	context.fillStyle = '#000';
 19 	context.fillRect(0, 0, width, height );
 20 
 21 	var image	= context.getImageData( 0, 0, canvas.width, canvas.height );
 22 	var imageData	= image.data;
 23 
 24 	var minColor	= new THREE.Color().setRGB(0.375, 0.125, 0);
 25 	var rngColor	= new THREE.Color().setRGB(0.5, 0.375, 0.375);
 26 
 27 	var normal	= new THREE.Vector3( 0, 0, 0 );
 28 	for ( var i = 0, j = 0, l = imageData.length; i < l; i += 4, j ++ ) {
 29 		// compute the normal
 30 		normal.x	= data[ j - 2 ] - data[ j + 2 ];
 31 		normal.y	= 2;
 32 		normal.z	= data[ j - width * 2 ] - data[ j + width * 2 ];
 33 		normal.normalize();
 34 		// compute the shade
 35 		var shade	= normal.dot( sun );
 36 		// fill the pixel
 37 		// - make color tunable
 38 		var factor		= 256.0 * ( 0.5 + data[ j ] * 0.015 );
 39 		//var factor		= 256.0;
 40 		imageData[i + 0]	= (minColor.r + shade * rngColor.r) * factor;
 41 		imageData[i + 1]	= (minColor.g + shade * rngColor.g) * factor;
 42 		imageData[i + 2]	= (minColor.b + shade * rngColor.b) * factor;
 43 	}
 44 	// draw the image in a canvas
 45 	context.putImageData(image, 0, 0);
 46 
 47 return canvas;
 48 
 49 	// Scaled 4x
 50 	var canvasScaled	= document.createElement( 'canvas' );
 51 	canvasScaled.width	= width  * 4;
 52 	canvasScaled.height	= height * 4;
 53 
 54 	var context	= canvasScaled.getContext( '2d' );
 55 	context.scale(4, 4);
 56 	context.drawImage(canvas, 0, 0);
 57 
 58 
 59 	var image	= context.getImageData(0, 0, canvasScaled.width, canvasScaled.height);
 60 	var imageData	= image.data;
 61 	for(var i = 0; i < imageData.length; i += 4){
 62 		var v	= Math.floor( Math.random() * 5 );
 63 		imageData[i + 0]	+= v;
 64 		imageData[i + 1]	+= v;
 65 		imageData[i + 2]	+= v;
 66 	}
 67 	context.putImageData(image, 0, 0);
 68 
 69 	return canvasScaled;
 70 
 71 }

 73 /**
 74  * Generate a terrain from Perlin
 75  * TODO make it more tunable
 76  * TODO make that usable standalone
 77 */
 78 function generateHeight( width, height )
 79 {
 80 	var size	= width * height;
 81 	var heights	= new Float32Array( size );
 82 	var perlin	= new ImprovedNoise();
 83 	var z		= Math.random() * 100;
 84 	var quality	= 1;
 85 
 86 	// zero the heights
 87 	for(var i = 0; i < size; i ++)	heights[i] = 0;
 88 
 89 // TODO find out what are all those constant and make them tunable
 90 
 91 	for(var j = 0; j < 4; j++ ){
 92 		for(var y = 0, i = 0; y < height; y++ ){
 93 			for(var x = 0; x < width; x++, i++ ){
 94 				var i		= x + y * width;
 95 				var noise	= perlin.noise( x / quality, y / quality, z );
 96 				heights[i]	+= Math.abs( noise * quality * 1.75 *2 );
 97 			}
 98 		}
 99 		quality *= 5;
100 	}
101 	// return the generated heights
102 	return heights;
103 }

111 tQuery.register('generateTerrainGeometry', function(opts){

113 	opts	= tQuery.extend(opts, {

115 		height		: 10,

117 		segmentsH	: 128*2
118 	});
119

121 	var tGeometry	= new THREE.PlaneGeometry(opts.width, opts.height, opts.segmentsW-1, opts.segmentsH-1 );
122 	tGeometry.dynamic= true;
123 
124 	var heights	= generateHeight( opts.segmentsW, opts.segmentsH );	
125 	for( var i = 0; i < tGeometry.vertices.length; i ++ ){
126 // TODO find out what are all those constant and make them tunable
127 		tGeometry.vertices[i].position.z = heights[i] / 375;
128 	}
129 	
130 	var canvas	= generateTexture( heights, opts.segmentsW, opts.segmentsH );
131 	var texture	= new THREE.Texture(canvas);
132 	texture.needsUpdate = true;
133 
134 	// mark the vertices as dirty
135 	tGeometry.__dirtyVertices = true;
136 	tGeometry.computeBoundingBox();
137 	tGeometry.computeCentroids();	
138 	tGeometry.computeFaceNormals();
139 	tGeometry.computeVertexNormals();
140 	
141 	return {
142 		geometry	: tQuery(tGeometry),
143 		texture		: texture
