  5 /**
  6  * Handle scene
  7  *
  8  * @class include THREE.Material
  9  *
 10  * @param {THREE.Material} object an instance or an array of instance
 11 */
 12 tQuery.Scene	= function()
 13 {
 14 	// update default scene.
 15 	// - TODO no sanity check ?
 16 	tQuery.scene	= this;
 17 	
 18 	// create a scene
 19 	this._scene	= new THREE.Scene();
 20 
 21 	// create a renderer
 22 	if( this._hasWebGL ){
 23 		this._renderer = new THREE.WebGLRenderer({
 24 			antialias		: true,	// to get smoother output
 25 			preserveDrawingBuffer	: true	// to allow screenshot
 26 		});
 27 		this._renderer.setClearColorHex( 0xBBBBBB, 1 );
 28 	}else{
 29 		this._addGetWebGLMessage();
 30 		throw new Error("WebGL required and not available")
 31 	}
 32 	// FIXME this window dimension is crap
 33 	this._renderer.setSize( window.innerWidth, window.innerHeight );
 34 
 35 	// create a camera in the scene
 36 	// FIXME this window dimension is crap
 37 	this._camera	= new THREE.PerspectiveCamera(35, window.innerWidth / window.innerHeight, 1, 10000 );
 38 	this._camera.position.set(0, 0, 5);
 39 	this._scene.add(this._camera);
 40 };
 41 
 42 // make it pluginable
 43 tQuery.pluginsMixin(tQuery.Scene);
 44

 46 tQuery.Scene.prototype.destroy	= function()
 47 {
 48 	// remove renderer element
 49 	var parent	= this._renderer.domElement.parentElement;
 50 	parent	&& parent.removeChild(this._renderer.domElement);
 51 }

 57 /**
 58  * true if webgl is available, false otherwise
 59 */
 60 tQuery.Scene.prototype._hasWebGL	= (function(){
 61 	// test from Detector.js
 62 	try{
 63 		return !! window.WebGLRenderingContext && !! document.createElement( 'canvas' ).getContext( 'experimental-webgl' );
 64 	} catch( e ){
 65 		return false;
 66 	}
 67 })();

 70 */
 71 tQuery.Scene.prototype._addGetWebGLMessage	= function(parent)
 72 {
 73 	parent	= parent || document.body;
 74 	
 75 	// message directly taken from Detector.js
 76 	var domElement = document.createElement( 'div' );
 77 	domElement.style.fontFamily = 'monospace';
 78 	domElement.style.fontSize = '13px';
 79 	domElement.style.textAlign = 'center';
 80 	domElement.style.background = '#eee';
 81 	domElement.style.color = '#000';
 82 	domElement.style.padding = '1em';
 83 	domElement.style.width = '475px';
 84 	domElement.style.margin = '5em auto 0';
 85 	domElement.innerHTML = window.WebGLRenderingContext ? [
 86 		'Your graphics card does not seem to support <a href="http://khronos.org/webgl/wiki/Getting_a_WebGL_Implementation">WebGL</a>.<br />',
 87 		'Find out how to get it <a href="http://get.webgl.org/">here</a>.'
 88 	].join( '\n' ) : [
 89 		'Your browser does not seem to support <a href="http://khronos.org/webgl/wiki/Getting_a_WebGL_Implementation">WebGL</a>.<br/>',
 90 		'Find out how to get it <a href="http://get.webgl.org/">here</a>.'
 91 	].join( '\n' );
 92 
 93 	parent.appendChild(domElement);
 94 }

100 /**
101  * add an object to the scene
102  * 
103  * @param {tQuery.Object3D} object3D to add to the scene (THREE.Object3D is accepted)
104 */
105 tQuery.Scene.prototype.add	= function(object3d)
106 {
107 	if( object3d instanceof tQuery.Object3D ){
108 		object3d.each(function(object3d){
109 			this._scene.add(object3d)			
110 		}.bind(this));
111 	}else if( object3d instanceof THREE.Object3D ){
112 		this._scene.add(object3d)		
113 	}else	console.assert(false, "invalid type");
114 	// for chained API
115 	return this;
116 }
117 
118 /**
119  * remove an object to the scene
120  * 
121  * @param {tQuery.Object3D} object3D to add to the scene (THREE.Object3D is accepted)
122 */
123 tQuery.Scene.prototype.remove	= function(object3d)
124 {

126 		object3d.each(function(object3d){
127 			this._scene.remove(object3d)
128 		}.bind(this));
129 	}else if( object3d instanceof THREE.Object3D ){
130 		this._scene.remove(object3d)
131 	}else	console.assert(false, "invalid type");
132 	// for chained API
133 	return this;
134 }

150 tQuery.Scene.prototype.appendTo	= function(domElement)
151 {
152 	domElement.appendChild(this._renderer.domElement)
153 	this._renderer.setSize( domElement.offsetWidth, domElement.offsetHeight );
154 	// for chained API
155 	return this;
156 }
