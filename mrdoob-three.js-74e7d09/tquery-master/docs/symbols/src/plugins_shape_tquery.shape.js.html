1 tQuery.register('Shape', function(tShape){
  2 	this._tShape	= tShape	|| new THREE.Shape();
  3 });
  4 
  5 //////////////////////////////////////////////////////////////////////////////////
  6 //										//
  7 //////////////////////////////////////////////////////////////////////////////////
  8 
  9 tQuery.Shape.prototype.extrude	= function(options){
 10 	// handle parameters
 11 	options	= tQuery.extend(options, tQuery.defaultExtrudeSettings);
 12 	// do the extrude
 13 	var tGeometry	= this._tShape.extrude( options );
 14 	// set the geometry.dynamic by default
 15 	tGeometry.dynamic= true;
 16 	// return the tQuery
 17 	return tQuery(tGeometry)
 18 }
 19 
 20 //////////////////////////////////////////////////////////////////////////////////
 21 //		forward functions directly to THREE.Shape			//
 22 //////////////////////////////////////////////////////////////////////////////////
 23 
 24 tQuery.Shape.forwardFns	= [
 25 	'moveTo',
 26 	'lineTo',
 27 	'bezierCurveTo',
 28 	'quadraticCurveTo'
 29 ];
 30 
 31 tQuery.Shape.forwardFns.forEach(function(fn){
 32 	tQuery.Shape.prototype[fn]	= function(){
 33 		// actually call the function
 34 		this._tShape[fn].apply(this._tShape, arguments)
 35 		// for chained API
 36 		return this;
 37 	}
 38 })
