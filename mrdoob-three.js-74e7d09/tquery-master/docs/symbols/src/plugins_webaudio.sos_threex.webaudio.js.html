 34 THREEx.WebAudio	= function(){
 35 	// create the context
 36 	this._ctx	= new webkitAudioContext();
 37 
 38 	// setup the end of the node chain
 39 	// TODO later code the clipping detection from http://www.html5rocks.com/en/tutorials/webaudio/games/ 
 40 	this._gainNode	= this._ctx.createGainNode();
 41 	this._compressor= this._ctx.createDynamicsCompressor();
 42 	this._gainNode.connect( this._compressor );
 43 	this._compressor.connect( this._ctx.destination );
 44 };

 80 THREEx.WebAudio.prototype.updateListener	= function(object3d, deltaTime){
 81 	var context	= this._ctx;
 82 	// sanity check on parameters
 83 	console.assert( object3d instanceof THREE.Object3D );
 84 	console.assert( typeof(deltaTime) === 'number' );
 85 
 86 	// ensure object3d.matrixWorld is up to date
 87 	object3d.updateMatrixWorld();
 88 	
 89 	////////////////////////////////////////////////////////////////////////
 90 	// set position
 91 	var position	= object3d.matrixWorld.getPosition();
 92 	context.listener.setPosition(position.x, position.y, position.z);
 93 
 94 	////////////////////////////////////////////////////////////////////////
 95 	// set orientation
 96 	var mOrientation= object3d.matrixWorld.clone();
 97 	// zero the translation
 98 	mOrientation.setPosition({x : 0, y: 0, z: 0});
 99 	// Compute Front vector: Multiply the 0,0,1 vector by the world matrix and normalize the result.
100 	var vFront= new THREE.Vector3(0,0,1);
101 	mOrientation.multiplyVector3(vFront);
102 	vFront.normalize();
103 	// Compute UP vector: Multiply the 0,-1,0 vector by the world matrix and normalize the result.
104 	var vUp= new THREE.Vector3(0,-1, 0);
105 	mOrientation.multiplyVector3(vUp);
106 	vUp.normalize();
107 	// Set panner orientation
108 	context.listener.setOrientation(vFront.x, vFront.y, vFront.z, vUp.x, vUp.y, vUp.z);
109 
110 	////////////////////////////////////////////////////////////////////////
111 	// set velocity
112 	if( this._prevPos === undefined ){
113 		this._prevPos	= object3d.matrixWorld.getPosition().clone();
114 	}else{
115 		var position	= object3d.matrixWorld.getPosition();
116 		var velocity	= position.clone().subSelf(this._prevPos).divideScalar(deltaTime);
117 		this._prevPos	= object3d.matrixWorld.getPosition().clone();
118 		context.listener.setVelocity(velocity.x, velocity.y, velocity.z);
119 	}
120 }

139 THREEx.WebAudio.Sound	= function(webaudio, nodeChain){
140 	console.assert( webaudio instanceof THREEx.WebAudio );
141 	this._webaudio	= webaudio;
142 	this._context	= webaudio.context();
143 	
144 	// create a default NodeChain if needed
145 	if( nodeChain === undefined ){
146 		nodeChain	= new THREEx.WebAudio.NodeChain(webaudio.context())
147 					.bufferSource().analyser().panner();
148 	}
149 	// setup this._chain
150 	console.assert( nodeChain instanceof THREEx.WebAudio.NodeChain );
151 	this._chain	= nodeChain;
152 
153 	// connect this._chain.last() node to this._webaudio._entryNode()
154 	this._chain.last().connect( this._webaudio._entryNode() );
155 	
156 	// create some alias
157 	this._source	= this._chain.nodes().bufferSource;
158 	this._analyser	= this._chain.nodes().analyser;
159 	this._panner	= this._chain.nodes().panner;
160 	
161 	// sanity check
162 	console.assert(this._source	, "no bufferSource: not yet supported")
163 	console.assert(this._analyser	, "no analyser: not yet supported")
164 	console.assert(this._panner	, "no panner: not yet supported")
165 };
166 
167 THREEx.WebAudio.Sound.prototype.destroy	= function(){
168 	// disconnect from this._webaudio
169 	this._chain.last().disconnect();
170 	// destroy this._chain
171 	this._chain.destroy();
172 	this._chain	= null;
173 };

224 /**
225  * Update the source with object3d. usefull for positional sounds
226  * 
227  * @param {THREE.Object3D} object3d the object which originate the source
228  * @param {Number} deltaTime the number of seconds since last update
229 */
230 THREEx.WebAudio.Sound.prototype.updateWithObject3d	= function(object3d, deltaTime){
231 	// sanity check on parameters
232 	console.assert( object3d instanceof THREE.Object3D );
233 	console.assert( typeof(deltaTime) === 'number' );
234 
235 	// ensure object3d.matrixWorld is up to date

237 	
238 	this.updateWithMatrix4(object3d.matrixWorld, deltaTime);
239 	
240 	return this;	// for chained API

243 /**
244  * Update the source with a matrixWorld. usefull for positional sounds
245  * 
246  * @param {THREE.Matrix4} matrixWorld the matrixWorld describing the position of the sound
247  * @param {Number} deltaTime the number of seconds since last update
248 */
249 THREEx.WebAudio.Sound.prototype.updateWithMatrix4	= function(matrixWorld, deltaTime){
250 	// sanity check on parameters
251 	console.assert( matrixWorld instanceof THREE.Matrix4 );
252 	console.assert( typeof(deltaTime) === 'number' );
253 
254 	////////////////////////////////////////////////////////////////////////
255 	// set position
256 	var position	= matrixWorld.getPosition();
257 	this._panner.setPosition(position.x, position.y, position.z);
258 
259 	////////////////////////////////////////////////////////////////////////
260 	// set orientation
261 	var vOrientation= new THREE.Vector3(0,0,1);
262 	var mOrientation= matrixWorld.clone();
263 	// zero the translation
264 	mOrientation.setPosition({x : 0, y: 0, z: 0});
265 	// Multiply the 0,0,1 vector by the world matrix and normalize the result.
266 	mOrientation.multiplyVector3(vOrientation);
267 	vOrientation.normalize();
268 	// Set panner orientation
269 	this._panner.setOrientation(vOrientation.x, vOrientation.y, vOrientation.z);
270 	
271 	////////////////////////////////////////////////////////////////////////
272 	// set velocity
273 	if( this._prevPos === undefined ){
274 		this._prevPos	= matrixWorld.getPosition().clone();
275 	}else{
276 		var position	= matrixWorld.getPosition();
277 		var velocity	= position.clone().subSelf(this._prevPos).divideScalar(deltaTime);
278 		this._prevPos	= matrixWorld.getPosition().clone();
279 		this._panner.setVelocity(velocity.x, velocity.y, velocity.z);
280 	}
281 }

287 /**
288  * Load a sound
289  *
290  * @param {String} url the url of the sound to load
291  * @param {Function} callback function to notify once the url is loaded (optional)
292 */
293 THREEx.WebAudio.Sound.prototype.load = function(url, callback){
294 	this._loadAndDecodeSound(url, function(buffer){
295 		this._source.buffer	= buffer;
296 		callback && callback(this);
297 	}.bind(this), function(){
298 		console.warn("unable to load sound "+url);
299 	});
300 	return this;	// for chained API
301 };
302 
303 /**
304  * Load and decode a sound
305  *
306  * @param {String} url the url where to get the sound
307  * @param {Function} onLoad the function called when the sound is loaded and decoded (optional)
308  * @param {Function} onError the function called when an error occured (optional)
309 */
310 THREEx.WebAudio.Sound.prototype._loadAndDecodeSound	= function(url, onLoad, onError){
311 	var context	= this._context;
312 	var request	= new XMLHttpRequest();
313 	request.open('GET', url, true);
314 	request.responseType	= 'arraybuffer';
315 	// Decode asynchronously
316 	request.onload	= function() {
317 		context.decodeAudioData(request.response, function(buffer) {
318 			onLoad && onLoad(buffer);
319 		}, function(){
320 			onError && onError();
321 		});
322 	};
323 	// actually start the request
324 	request.send();
325 }

362 THREEx.WebAudio.NodeChain.prototype._addNode	= function(node, properties)
363 {
364 	// connect this._lastNode to node if suitable
365 	if( this._lastNode !== null )	this._lastNode.connect(node);
366 	
367 	// update this._firstNode && this._lastNode
368 	if( this._firstNode === null )	this._firstNode	= node;
369 	this._lastNode	= node;
370 	
371 	// apply properties to the node
372 	for( var property in properties ){
373 		node[property]	= properties[property];
374 	}
375 
376 	// for chained API
377 	return this;
378 };

380 THREEx.WebAudio.NodeChain.prototype.bufferSource	= function(properties){
381 	var node		= this._context.createBufferSource()
382 	this._nodes.bufferSource= node;
