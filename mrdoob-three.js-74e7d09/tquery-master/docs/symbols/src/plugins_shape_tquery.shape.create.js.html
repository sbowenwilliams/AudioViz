 71 tQuery.register('createRectangleShape', function(width, height){
 72 	var w		= width !== undefined ? width : 80;
 73 	var h		= height!== undefined ? height: 80;
 74 	var shape	= tQuery.createShape()
 75 		.moveTo(  0, 0 )
 76 		.lineTo(  0, h )
 77 		.lineTo(  w, h )
 78 		.lineTo(  w, 0 )	
 79 		.lineTo(  0, 0 );	
 80 	return shape;
 81 });

 83 tQuery.register('createCircleShape', function(radius){
 84 	radius	= radius !== undefined ? radius	: 40;
 85 	// TODO normalize it
 86 	var shape	= tQuery.createShape()
 87 		.moveTo( 0, radius )
 88 		.quadraticCurveTo(  radius,  radius,  radius, 0 )
 89 		.quadraticCurveTo(  radius, -radius, 0, -radius )
 90 		.quadraticCurveTo( -radius, -radius, -radius, 0 )
 91 		.quadraticCurveTo( -radius,  radius, 0,  radius );
 92 	return shape;

109 	var arcShape	= new THREE.Shape();
110 	arcShape.moveTo( 0, 0 );
111 	arcShape.arc( 10, 10, 30, 0, Math.PI*2, false );
112 	
113 	var holePath	= new THREE.Path();
114 	holePath.moveTo( 0, 0 );
115 	holePath.arc( 10, 10, 10, 0, Math.PI*2, true );
116 
117 	arcShape.holes.push( holePath );
118 
119 	return new tQuery.Shape(arcShape);
