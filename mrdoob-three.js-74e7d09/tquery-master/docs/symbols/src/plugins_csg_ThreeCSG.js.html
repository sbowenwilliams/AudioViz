 11 THREE.CSG = {
 12 	toCSG: function ( three_model, offset, rotation ) {
 13 		var i, geometry, offset, polygons, vertices, rotation_matrix;
 14 		
 15 		if ( !CSG ) {
 16 			throw 'CSG library not loaded. Please get a copy from https://github.com/evanw/csg.js';
 17 		}
 18 		
 19 		if ( three_model instanceof THREE.Mesh ) {
 20 			geometry = three_model.geometry;
 21 			offset = offset || three_model.position;
 22 			rotation = rotation || three_model.rotation;
 23 		} else if ( three_model instanceof THREE.Geometry ) {
 24 			geometry = three_model;
 25 			offset = offset || new THREE.Vector3( 0, 0, 0 );
 26 			rotation = rotation || new THREE.Vector3( 0, 0, 0 );
 27 		} else {
 28 			throw 'Model type not supported.';
 29 		}
 30 		rotation_matrix = new THREE.Matrix4( ).setRotationFromEuler( rotation );
 31 		
 32 		var polygons = [];
 33 		for ( i = 0; i < geometry.faces.length; i++ ) {
 34 			if ( geometry.faces[i] instanceof THREE.Face3 ) {
 35 				
 36 				
 37 				vertices = [];
 38 				vertices.push( new CSG.Vertex( rotation_matrix.multiplyVector3( geometry.vertices[geometry.faces[i].a].position.clone( ).addSelf( offset ) ), [ geometry.faces[i].normal.x, geometry.faces[i].normal.y, geometry.faces[i].normal.z ] ) );
 39 				vertices.push( new CSG.Vertex( rotation_matrix.multiplyVector3( geometry.vertices[geometry.faces[i].b].position.clone( ).addSelf( offset ) ), [ geometry.faces[i].normal.x, geometry.faces[i].normal.y, geometry.faces[i].normal.z ] ) );
 40 				vertices.push( new CSG.Vertex( rotation_matrix.multiplyVector3( geometry.vertices[geometry.faces[i].c].position.clone( ).addSelf( offset ) ), [ geometry.faces[i].normal.x, geometry.faces[i].normal.y, geometry.faces[i].normal.z ] ) );
 41 				polygons.push( new CSG.Polygon( vertices ) );
 42 				
 43 			} else if ( geometry.faces[i] instanceof THREE.Face4 ) {
 44 				
 45 				vertices = [];
 46 				vertices.push( new CSG.Vertex( rotation_matrix.multiplyVector3( geometry.vertices[geometry.faces[i].a].position.clone( ).addSelf( offset ) ), [ geometry.faces[i].normal.x, geometry.faces[i].normal.y, geometry.faces[i].normal.z ] ) );
 47 				vertices.push( new CSG.Vertex( rotation_matrix.multiplyVector3( geometry.vertices[geometry.faces[i].b].position.clone( ).addSelf( offset ) ), [ geometry.faces[i].normal.x, geometry.faces[i].normal.y, geometry.faces[i].normal.z ] ) );
 48 				vertices.push( new CSG.Vertex( rotation_matrix.multiplyVector3( geometry.vertices[geometry.faces[i].d].position.clone( ).addSelf( offset ) ), [ geometry.faces[i].normal.x, geometry.faces[i].normal.y, geometry.faces[i].normal.z ] ) );
 49 				polygons.push( new CSG.Polygon( vertices ) );
 50 				
 51 				vertices = [];
 52 				vertices.push( new CSG.Vertex( rotation_matrix.multiplyVector3( geometry.vertices[geometry.faces[i].b].position.clone( ).addSelf( offset ) ), [ geometry.faces[i].normal.x, geometry.faces[i].normal.y, geometry.faces[i].normal.z ] ) );
 53 				vertices.push( new CSG.Vertex( rotation_matrix.multiplyVector3( geometry.vertices[geometry.faces[i].c].position.clone( ).addSelf( offset ) ), [ geometry.faces[i].normal.x, geometry.faces[i].normal.y, geometry.faces[i].normal.z ] ) );
 54 				vertices.push( new CSG.Vertex( rotation_matrix.multiplyVector3( geometry.vertices[geometry.faces[i].d].position.clone( ).addSelf( offset ) ), [ geometry.faces[i].normal.x, geometry.faces[i].normal.y, geometry.faces[i].normal.z ] ) );
 55 				polygons.push( new CSG.Polygon( vertices ) );
 56 				
 57 			} else {
 58 				throw 'Model contains unsupported face.';
 59 			}
 60 		}
 61 		
 62 		return CSG.fromPolygons( polygons );
 63 	},
 64 	
 65 	fromCSG: function( csg_model ) {
 66 		var i, j, vertices, face,
 67 			three_geometry = new THREE.Geometry( ),
 68 			polygons = csg_model.toPolygons( );
 69 		
 70 		if ( !CSG ) {
 71 			throw 'CSG library not loaded. Please get a copy from https://github.com/evanw/csg.js';
 72 		}
 73 		
 74 		for ( i = 0; i < polygons.length; i++ ) {
 75 			
 76 			// Vertices
 77 			vertices = [];
 78 			for ( j = 0; j < polygons[i].vertices.length; j++ ) {
 79 				vertices.push( this.getGeometryVertice( three_geometry, polygons[i].vertices[j].pos ) );
 80 			}
 81 			if ( vertices[0] === vertices[vertices.length - 1] ) {
 82 				vertices.pop( );
 83 			}
 84 			
 85 			for (var j = 2; j < vertices.length; j++) {
 86 				face = new THREE.Face3( vertices[0], vertices[j-1], vertices[j], new THREE.Vector3( ).copy( polygons[i].plane.normal ) );
 87 				three_geometry.faces.push( face );
 88 				three_geometry.faceVertexUvs[0].push( new THREE.UV( ) );
 89 			}
 90 		}
 91 		
 92 		three_geometry.computeBoundingBox();
 93 		
 94 		return three_geometry;
 95 	},
 96

109 		return geometry.vertices.length - 1;
110 	}
