  7 tQuery.World.register('boilerplate', function(opts){
  8 	// put renderer fullpage
  9 	var domElement	= document.body;
 10 	domElement.style.margin		= "0";
 11 	domElement.style.padding	= "0";
 12 	domElement.style.overflow	= 'hidden';
 13 	this.appendTo(domElement);
 14 	this._renderer.setSize( domElement.offsetWidth, domElement.offsetHeight );
 15 	
 16 	// add the boilerplate
 17 	this.addBoilerplate(opts);
 18 	
 19 	// for chained API
 20 	return this;
 21 });

 23 tQuery.World.register('addBoilerplate', function(opts){
 24 	var _this	= this;
 25 	// sanity check - no boilerplate is already installed
 26 	console.assert( this.hasBoilerplate() !== true );
 27 	// handle parameters	
 28 	opts	= tQuery.extend(opts, {
 29 		honorInfo	: true,
 30 		stats		: true,
 31 		cameraControls	: true,
 32 		windowResize	: true,
 33 		screenshot	: true,
 34 		fullscreen	: true
 35 	});
 36 	// get the context
 37 	var ctx	= {};
 38 
 39 	// create the context
 40 	tQuery.data(this, '_boilerplateCtx', ctx);
 41 
 42 	// add css for the info element if any
 43 	if( opts.honorInfo ){
 44 		var element	= document.getElementById('info');
 45 		if( element ){
 46 			element.style.position	= "absolute";
 47 			element.style.width	= "100%";
 48 			element.style.textAlign	= "center";
 49 		}
 50 	}
 51 
 52 	// add Stats.js - https://github.com/mrdoob/stats.js
 53 	if( opts.stats ){
 54 		ctx.stats	= new Stats();
 55 		ctx.stats.domElement.style.position	= 'absolute';
 56 		ctx.stats.domElement.style.bottom	= '0px';
 57 		document.body.appendChild( ctx.stats.domElement );
 58 		ctx.loopStats	= function(){
 59 			ctx.stats.update();
 60 		};
 61 		this.loop().hook(ctx.loopStats);		
 62 	}
 63 
 64 	// get some variables
 65 	var camera	= this.camera();
 66 	var renderer	= this.renderer();
 67 
 68 	// create a camera contol
 69 	if( opts.cameraControls ){
 70 		ctx.cameraControls	= new THREEx.DragPanControls(camera);
 71 		this.setCameraControls(ctx.cameraControls);		
 72 	}
 73 
 74 	// transparently support window resize
 75 	if( opts.windowResize ){
 76 		ctx.windowResize	= THREEx.WindowResize.bind(renderer, camera);		
 77 	}
 78 	// allow 'p' to make screenshot
 79 	if( opts.screenshot ){		
 80 		ctx.screenshot		= THREEx.Screenshot.bindKey(renderer);
 81 	}
 82 	// allow 'f' to go fullscreen where this feature is supported
 83 	if( opts.fullscreen && THREEx.FullScreen.available() ){
 84 		ctx.fullscreen	= THREEx.FullScreen.bindKey();		
 85 	}
 86 
 87 	// bind 'destroy' event on tQuery.world
 88 	ctx._$onDestroy	= this.bind('destroy', function(){
 89 		if( this.hasBoilerplate() === false )	return;
 90 		this.removeBoilerplate();	
 91 	});
 92 	
 93 	// for chained API
 94 	return this;
 95 });

 97 tQuery.World.register('hasBoilerplate', function(){
 98 	// get the context
 99 	var ctx	= tQuery.data(this, "_boilerplateCtx")
100 	// return true if ctx if defined, false otherwise
101 	return ctx === undefined ? false : true;
102 });
103 
104 tQuery.World.register('removeBoilerplate', function(){
105 	// get context
106 	var ctx	= tQuery.data(this, '_boilerplateCtx');
107 	// if not present, return now
108 	if( ctx === undefined )	return	this;
109 	// remove the context from this
110 	tQuery.removeData(this, '_boilerplateCtx');
111 
112 	// unbind 'destroy' for tQuery.World
113 	this.unbind('destroy', this._$onDestroy);
114 
115 	// remove stats.js
116 	ctx.stats		&& document.body.removeChild(ctx.stats.domElement );
117 	ctx.stats		&& this.loop().unhook(ctx.loopStats);
118 	// remove camera
119 	ctx.cameraControls	&& this.removeCameraControls()
120 	// stop windowResize
121 	ctx.windowResize	&& ctx.windowResize.stop();
122 	// unbind screenshot
123 	ctx.screenshot		&& ctx.screenshot.unbind();
124 	// unbind fullscreen
125 	ctx.fullscreen		&& ctx.fullscreen.unbind();
126 });
