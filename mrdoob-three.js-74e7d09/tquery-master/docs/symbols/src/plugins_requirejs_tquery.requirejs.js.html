1 /**
  2  * Get the script base url from inside the script itself
  3  * https://github.com/jrburke/requirejs/wiki/Differences-between-the-simplified-CommonJS-wrapper-and-standard-AMD-define#wiki-module
  4  *
  5  * @param {Object} module is the magic module from requirejs
  6  * @returns {string} the base url
  7 */
  8 tQuery.register('scriptBaseUrl', function(module){
  9 	// paramter sanity check 
 10 	console.assert( Object.keys(module).length === 3 );
 11 	console.assert( module.id !== undefined && module.uri !== undefined );
 12 	// determine the baseUrl
 13 	var baseUrl	= location.protocol + "//" + location.host + location.pathname + module.uri;
 14 	baseUrl		= baseUrl.substr(0, baseUrl.lastIndexOf('/')+1);
 15 	return baseUrl;
 16 });
