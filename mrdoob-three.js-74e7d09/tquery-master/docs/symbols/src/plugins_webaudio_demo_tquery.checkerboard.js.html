1 /**
  2  * Create a checkerboard tQuery.Mesh
  3 */
  4 tQuery.register('createCheckerboard', function(opts){
  5 	// handle parameters
  6 	opts	= tQuery.extend(opts, {
  7 		width		: 1,		
  8 		height		: 1,
  9 		segmentsW	: 8,
 10 		segmentsH	: 8,
 11 		materialEven	: new THREE.MeshBasicMaterial({ color: 0xcccccc }),
 12 		materialOdd	: new THREE.MeshBasicMaterial({ color: 0x444444 })
 13 	});
 14 	// create the geometry	
 15 	var geometry		= new THREE.PlaneGeometry( opts.width, opts.height, opts.segmentsW, opts.segmentsH );
 16 	// set materials per faces
 17 	geometry.materials	= [opts.materialEven, opts.materialOdd];
 18 	geometry.faces.forEach(function(face, idx){
 19 		var y	= Math.floor(idx / opts.segmentsW);
 20 		var x	= idx - (y*opts.segmentsW);
 21 		face.materialIndex	= (y % 2 + x%2 ) %2;
 22 	});
 23 	// create the mesh
 24 	var material	= new THREE.MeshFaceMaterial();
 25 	var mesh	= new THREE.Mesh(geometry, material);
 26 	// return the tQuery
 27 	return tQuery(mesh);
 28 });
