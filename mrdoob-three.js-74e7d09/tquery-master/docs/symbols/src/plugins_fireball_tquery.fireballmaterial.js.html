  7 tQuery.Object3D.register('useFileballMaterial', function(scale){
  8 	scale	= scale !== undefined ? scale : 1;
  9 
 10 	this.each(function(object3d){
 11 		var uniforms	= {
 12 			time	: { type: "f", value: 1.0 },
 13 			scale	: { type: "f", value: scale }
 14 		};
 15 		
 16 		var material	= new THREE.ShaderMaterial({
 17 			uniforms	: uniforms,
 18 			vertexShader	: tQuery.Object3D.prototype.useFileballMaterial._vertexShaderText,
 19 			fragmentShader	: tQuery.Object3D.prototype.useFileballMaterial._fragmentShaderText,
 20 		});
 21 		
 22 		object3d.material	= material;
 23 		
 24 		tQuery.world.loop().hook(function(deltaTime, present){
 25 			uniforms.time.value += 0.275 * deltaTime;
 26 		});
 27 	});
 28 	// for chained API
 29 	return this;
 30 });
