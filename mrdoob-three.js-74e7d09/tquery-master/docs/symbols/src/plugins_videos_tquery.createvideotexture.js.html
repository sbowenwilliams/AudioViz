  4 tQuery.register('createVideoTexture', function(opts){
  5 	// handle parameters
  6 	if( arguments.length === 1 && typeof opts === 'string' ){
  7 		opts	= { url : arguments[0] };
  8 	}
  9 	opts	= tQuery.extend(opts, {
 10 		loop	: tQuery.world.loop()
 11 	});
 12 	console.assert( opts.url, "url MUST be specified" )
 13 
 14 	// create the video element
 15 	var video	= document.createElement('video');
 16 	video.width	= 320;
 17 	video.height	= 240;
 18 	video.autoplay	= true;
 19 	video.loop	= true;
 20 	video.src	= opts.url;
 21 
 22 	// create the texture
 23 	var texture	= new THREE.Texture( video );
 24 	
 25 	opts.loop.hook(function(){
 26 		if( video.readyState === video.HAVE_ENOUGH_DATA ){
 27 			texture.needsUpdate	= true;
 28 		}
 29 	});
 30 	
 31 	return texture;
 32 });
