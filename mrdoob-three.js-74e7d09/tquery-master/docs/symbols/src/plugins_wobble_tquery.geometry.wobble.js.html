1 /**
  2  * tquery.js plugin to handle keyboard
  3 */
  4 define(['threex/THREEx.GeometryWobble'], function(){	
  5 	tQuery.Geometry.register('wobble', function(){
  6 		this.each(function(geometry){
  7 			THREEx.GeometryWobble.init(geometry);
  8 			THREEx.GeometryWobble.cpuAxis(geometry, 'x', 4);
  9 	
 10 			tQuery.world.loop().hook(function(delta, present){
 11 				var piSecond	= present * Math.PI;
 12 				var phase	= 200 * piSecond / 180;
 13 				var magnitude	= 0.25;
 14 				THREEx.GeometryWobble.Animate(geometry, phase, magnitude);
 15 			});
 16 		});
 17 		// for chained API
 18 		return this;
 19 	});	
 20 });
