1 tQuery.Geometry.register('vertexAnimation', function(options){
  2 	// handle parameters
  3 	options	= tQuery.extend(options, {
  4 		transform	: function(o, v, time){
  5 			var a	= time*2 + v.y * 10;
  6 			v.x	= o.x + Math.cos(a)*0.1;
  7 		}
  8 	});	
  9 	var transform	= options.transform;
 10 
 11 	// init the animation
 12 	this.each(function(tGeometry){
 13 		var length	= tGeometry.vertices.length;
 14 		// TODO put that in a .data()
 15 		tGeometry._origVertices	= new Array(length)
 16 		for(var i = 0; i < length; i++){
 17 			var vertex	= new THREE.Vertex();
 18 			vertex.position.copy(tGeometry.vertices[i].position.clone())
 19 			tGeometry._origVertices[i]	= vertex;
 20 		}
 21 	});
 22 
 23 
 24 	// do the actual animation
 25 	tQuery.world.loop().hook(function(deltaTime, present){
 26 		this.each(function(tGeometry){
 27 			for(var i = 0; i < tGeometry.vertices.length; i++) {
 28 				var origVector3	= tGeometry._origVertices[i].position;
 29 				var vector3	= tGeometry.vertices[i].position;
 30 				transform(origVector3, vector3, present);
 31 			}
 32 			// mark the vertices as dirty
 33 			tGeometry.__dirtyVertices = true;
 34 			tGeometry.computeBoundingBox();
 35 			tGeometry.computeCentroids();
 36 			tGeometry.computeFaceNormals();
 37 			tGeometry.computeVertexNormals();
 38 			tGeometry.computeTangents();
 39 		});
 40 	}.bind(this));
 41 });
