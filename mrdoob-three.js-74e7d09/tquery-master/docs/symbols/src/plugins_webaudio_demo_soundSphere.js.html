1 var SoundSphere	= function(webaudio, soundUrl){	
  2 	var group	= tQuery.createObject3D();
  3 	this._group	= group;
  4 
  5 	var material	= new THREE.MeshLambertMaterial({
  6 		ambient	: 0x444444,
  7 		color	: 0xFF5588,
  8 		specular: 0xCC88ff,
  9 		shininess: 400
 10 	});
 11 
 12 	var post	= tQuery.createCube(0.1,1.5,0.1, material).addTo(group)
 13 		.translateY(1.5/2);
 14 	var sphere	= tQuery.createSphere(0.5, 32, 16, material).addTo(group)
 15 		.translateY(1.5);
 16 
 17 
 18 	// setup .castShadow	
 19 	post.get(0).castShadow	= true;
 20 	sphere.get(0).castShadow= true;
 21 	
 22 	// create the webaudio sound
 23 	var sound	= new tQuery.WebAudio.Sound(webaudio);
 24 	sound.load(soundUrl, function(sound){
 25 		sound._source.loop	= true;
 26 		//sound.volume(0.5);
 27 		sound.play();
 28 	});
 29 	// update sound position with this._group position
 30 	world.loop().hook(function(deltaTime){
 31 		var object3d	= this._group.get(0);
 32 		sound.updateWithObject3d(object3d, deltaTime)
 33 	}.bind(this));
 34 
 35 	this._initSoundScale(sound, sphere);
 36 };
 37 
 38 SoundSphere.prototype.object3D	= function(){
 39 	return this._group;
 40 }

 42 SoundSphere.prototype._initSoundScale	= function(sound, object){
 43 	var analyser	= sound._analyser;
 44 	var freqByte	= new Uint8Array(analyser.frequencyBinCount);
 45 	analyser.smoothingTimeConstant = 0.6;
 46 
 47 	var minColor	= new THREE.Color(0x1111FF);
 48 	var maxColor	= new THREE.Color(0xFFaaFF);
 49 
 50 	tQuery.world.loop().hook(function(){
 51 		analyser.getByteFrequencyData(freqByte);
 52 		var nb		= 2;
 53 		var total	= 0;
 54 		for(var i = 0; i < nb; i++){
 55 			total	+= freqByte[i];
 56 		}
 57 		var scale	= total / (nb*256-1);
 58 		console.assert(scale >= 0.0);
 59 		console.assert(scale <= 1.0);
 60 
 61 		object.scale(scale*1.5+0.5);
 62 
 63 		var colorScale	= scale*1;
 64 		colorScale	= Math.max(colorScale, 0)
 65 		colorScale	= Math.min(colorScale, 1)
 66 		var colorR	= colorScale * (maxColor.r - minColor.r) + minColor.r;
 67 		var colorG	= colorScale * (maxColor.g - minColor.g) + minColor.g;
 68 		var colorB	= colorScale * (maxColor.b - minColor.b) + minColor.b;
 69 		object.get(0).material.color.setRGB(colorR, colorG, colorB);
 70 	})
 71 }
