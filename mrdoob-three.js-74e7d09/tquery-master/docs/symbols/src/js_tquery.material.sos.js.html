1 /**
  2  * Handle material
  3  *
  4  * @class include THREE.Material
  5  *
  6  * @param {THREE.Material} object an instance or an array of instance
  7 */
  8 tQuery.Material	= function(object){
  9 	this._lists	= object instanceof Array ? object : [object];
 10 	this.length	= this._lists.length;
 11 	// sanity check - all items MUST be THREE.Geometery
 12 	this._lists.forEach(function(item){ console.assert(item instanceof THREE.Material); });
 13 };
 14 
 15 // Make it pluginable
 16 tQuery.pluginsMixin(tQuery.Material, tQuery.Material.prototype);

 23 /**
 24  * Retrieve the elements matched by the jQuery object
 25  * 
 26  * @param {Function} callback the function to notify. function(element){ }.
 27  * 			loop interrupted if it returns false
 28  * 
 29  * @returns {Boolean} return true if completed, false if interrupted
 30 */
 31 tQuery.Material.prototype.get	= function(idx){
 32 	if( idx === undefined )	return this._lists;
 33 
 34 	// sanity check - it MUST be defined
 35 	console.assert(this._lists[idx], "element not defined");
 36 	return this._lists[idx];
 37 };
 38 
 39 /**
 40  * loop over element
 41  * 
 42  * @param {Function} callback the function to notify. function(element){ }.
 43  * 			loop interrupted if it returns false
 44  * 
 45  * @returns {Boolean} return true if completed, false if interrupted
 46 */
 47 tQuery.Material.prototype.each	= function(callback){
 48 	for(var i = 0; i < this._lists.length; i++){
 49 		var object3d	= this._lists[i];
 50 		var keepLooping	= callback(object3d)
 51 		if( keepLooping === false )	return false;
 52 	}
 53 	return true;
 54 };
 55 
 56 /**
 57  * getter/setter of the back pointer
 58  *
 59  * @param {Object} back the value to return when .back() is called. optional
 60  * 
 61 */
 62 tQuery.Material.prototype.back	= function(back){
 63 	if( back === undefined )	return this._back;
 64 	this._back	= back;
 65 	return this;
 66 };
