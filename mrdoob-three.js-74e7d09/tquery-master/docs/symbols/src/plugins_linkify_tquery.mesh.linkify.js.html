1 /**
  2  * plugin to handle deviceOrientation API
  3 */
  4 define(['plugins/csg/tquery.object3d.csg'], function(){
  5 	tQuery.Mesh.register('linkify', function(url){
  6 		var mesh	= tQuery(this.get(0));
  7 		// measure mesh size
  8 		var size	= mesh.geometry().computeAll().size();
  9 		
 10 		// build the underline
 11 		var underlineH	= size.y / 10;
 12 		var deltaY	= size.y / 20;
 13 		var underline	= tQuery.createCube(size.x, underlineH, size.z)
 14 				.translateY(-size.y/2 - deltaY - underlineH/2);
 15 		// make it invisible by default
 16 		underline.get(0).visible	= false;
 17 		// add it to the mesh
 18 		mesh.add(underline);
 19 		
 20 		// bind the click
 21 		mesh.on('click', function(event){
 22 			window.open(url, '_blank');
 23 		});
 24 		// bind 'mouseover'
 25 		mesh.on('mouseover', function(){
 26 			underline.get(0).visible	= true;
 27 			document.body.style.cursor	= 'pointer';
 28 		});
 29 		// bind 'mouseout'
 30 		mesh.on('mouseout', function(){
 31 			underline.get(0).visible	= false;		
 32 			document.body.style.cursor	= 'default';
 33 		});
 34 	});
 35 });
