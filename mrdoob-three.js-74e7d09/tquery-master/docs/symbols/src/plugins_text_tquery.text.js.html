1 /**
  2  * tQuery.createText('foobar')
  3 */
  4 define(function(){
  5 	tQuery.register('createText', function(text, options){
  6 		// handle parameters
  7 		options	= tQuery.extend(options, {
  8 			size		: 1,
  9 			height		: 0.4,
 10 			curveSegments	: 4,
 11 	
 12 			weight		: "bold",
 13 			font		: "droid serif"
 14 		});
 15 		// create the geometry
 16 		var geometry	= new THREE.TextGeometry(text, options);
 17 	
 18 		// center the geometry
 19 		// - THREE.TextGeometry isnt centered for unknown reasons. all other geometries are centered
 20 		geometry.computeBoundingBox();
 21 		var center	= new THREE.Vector3();
 22 		center.x	= ( geometry.boundingBox.max.x - geometry.boundingBox.min.x ) / 2;
 23 		center.y	= ( geometry.boundingBox.max.y - geometry.boundingBox.min.y ) / 2;
 24 		center.z	= ( geometry.boundingBox.max.z - geometry.boundingBox.min.z ) / 2;
 25 		geometry.vertices.forEach(function(vertex){
 26 			vertex.position.subSelf(center);
 27 		});
 28 		
 29 		// create a mesh with it
 30 		var material	= new THREE.MeshNormalMaterial();
 31 		var mesh	= new THREE.Mesh( geometry, material );
 32 		// return a tQuery object
 33 		return tQuery(mesh);
 34 	});
 35 });
