  5 /**
  6  * add a THREE.FogExp2 to this world.scene()
  7  *
  8  * @param opts the options
  9  * @param opts.colorHex the hexa value of the fog color
 10  * @param opts.density the density of the fog
 11 */
 12 tQuery.World.register('addFogExp2', function(opts){
 13 	// handle parameter
 14 	opts	= tQuery.extend(opts, {
 15 		colorHex	: this.renderer().getClearColor().getHex(),
 16 		density		: 0.1
 17 	});
 18 	// set the fog
 19 	this.scene().fog	= new THREE.FogExp2(opts.colorHex, opts.density );
 20 	// for chained API
 21 	return this;
 22 });
 23 
 24 /**
 25  * add a THREE.Fog to this world.scene()
 26  *
 27  * @param opts the options
 28  * @param opts.colorHex the hexa value of the fog color
 29  * @param opts.near how far away from camera, the fog starts
 30  * @param opts.far how far away from camera, the fog ends
 31 */
 32 tQuery.World.register('addFog', function(opts){
 33 	// handle parameter
 34 	opts	= tQuery.extend(opts, {
 35 		colorHex	: this.renderer().getClearColor().getHex(),
 36 		near		: 1,
 37 		far		: 20
 38 	});
 39 	// set the fog
 40 	this.scene().fog	= new THREE.Fog(opts.colorHex, opts.near, opts.far);
 41 	// for chained API
 42 	return this;
 43 });
 44 
 45 
 46 /**
 47  * remove fog from this world.scene()
 48 */
 49 tQuery.World.register('removeFog', function(){
 50 	this.scene().fog= null;
 51 	console.assert(false, "doesnt work");
 52 	// for chained API
 53 	return this;
 54 });
