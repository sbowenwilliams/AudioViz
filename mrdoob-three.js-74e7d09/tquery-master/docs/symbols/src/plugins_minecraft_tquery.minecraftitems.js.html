1 tQuery.register('MinecraftItems', function(opts){
  2 	// handle parameters
  3 	this._opts	= tQuery.extend(opts, {
  4 		url	: 'images/items/items.png'
  5 	});
  6 	function getMaterial(image, transparent){
  7 		var tex		= new THREE.Texture(image);
  8 		tex.magFilter	= THREE.NearestFilter;
  9 		tex.minFilter	= THREE.NearestFilter;
 10 		tex.format	= transparent ? THREE.RGBAFormat : THREE.RGBFormat;
 11 		tex.needsUpdate	= true;
 12 		var material	= new THREE.MeshBasicMaterial({
 13 			map		: tex,
 14 			transparent	: transparent ? true : false
 15 		});
 16 		return material;
 17 	}
 18 	function uvmap (geometry, face, x, y, w, h, rotateBy){
 19 		rotateBy	= rotateBy !== undefined ? rotateBy : 0;
 20 		var uvs		= geometry.faceVertexUvs[0][face];
 21 		var tileU	= x;
 22 		var tileV	= y;
 23 		
 24 		uvs[ (0 + rotateBy) % 4 ].u = tileU * tileUvW;
 25 		uvs[ (0 + rotateBy) % 4 ].v = tileV * tileUvH;
 26 		uvs[ (1 + rotateBy) % 4 ].u = tileU * tileUvW;
 27 		uvs[ (1 + rotateBy) % 4 ].v = tileV * tileUvH + h * tileUvH;
 28 		uvs[ (2 + rotateBy) % 4 ].u = tileU * tileUvW + w * tileUvW;
 29 		uvs[ (2 + rotateBy) % 4 ].v = tileV * tileUvH + h * tileUvH;
 30 		uvs[ (3 + rotateBy) % 4 ].u = tileU * tileUvW + w * tileUvW;
 31 		uvs[ (3 + rotateBy) % 4 ].v = tileV * tileUvH;
 32 	};
 33 	/**
 34 	 * Create the geometry of the sprite 'id'. It is cached
 35 	 * @returns {THREE.Geometry} geometry built
 36 	*/
 37 	function getGeometry( id ){
 38 		if( geometries[id] !== undefined )	return geometries[id];
 39 		
 40 		function getSides(x, y){
 41 			var ix = Math.floor(id % 16)*16;
 42 			var iy = Math.floor(id / 16)*16;
 43 			
 44 			var alphaPx	= (x+1) < 16 ? imd[((x+1)+y*16)*4+3] : 0;
 45 			var alphaNx	= (x-1) >= 0 ? imd[((x-1)+y*16)*4+3] : 0;
 46 			var alphaPy	= (y+1) < 16 ? imd[(x+(y-1)*16)*4+3] : 0;
 47 			var alphaNy	= (y-1) >= 0 ? imd[(x+(y+1)*16)*4+3] : 0;
 48 			
 49 			return {
 50 				px: !alphaPx, // Turns zero and undefined to true
 51 				nx: !alphaNx,
 52 				py: !alphaPy,
 53 				ny: !alphaNy,
 54 				pz: true,
 55 				nz: true
 56 			};
 57 		};
 58 
 59 		var imgdata	= context.getImageData(Math.floor(id % 16)*16, Math.floor(id / 16)*16, 16, 16);
 60 		var imd		= imgdata.data;			
 61 		var geometry	= new THREE.Geometry();
 62 		
 63 		for(var x=0; x < 16; x++) {
 64 			for(var y=0; y < 16; y++) {
 65 				// is this pixel transparent, skip it
 66 				if( imd[(x+y*16)*4+3] === 0)	continue;
 67 				
 68 				var voxel	= new THREE.CubeGeometry(1, 1, 2, 1, 1, 1, undefined, getSides(x, y));
 69 				// TODO why is there a texture here ????
 70 				// - this is a single pixel. get the pixel and set the color
 71 				// - this may fix the anti alias issue
 72 				for(var i=0; i < voxel.faceVertexUvs[0].length; i++) { // Fix color of voxel
 73 					uvmap(voxel, i, Math.floor(id % 16)*16+x, Math.floor(id / 16)*16+y, 1, 1);
 74 				}
 75 				
 76 				// TODO what is this ... it seems a translation but why ?
 77 				console.assert(voxel.vertices.length)
 78 				for(var i=0; i < voxel.vertices.length; i++) { // Fix voxel's position
 79 					voxel.vertices[i].x += +(x-15/2);	// what is this 7.5 ? why not 8 ? as in 16/2
 80 					voxel.vertices[i].y += -(y-15/2);
 81 				}
 82 				THREE.GeometryUtils.merge(geometry, voxel);
 83 			}
 84 		}
 85 		// if the geometry is fully transparent, unset it - NOTE: this is set to null, not unsigned
 86 		if( geometry.faces.length === 0 )	geometry = null;
 87 		// cache the result
 88 		geometries[id]	= geometry;
 89 		// return the result
 90 		return geometry;
 91 	};
 92 	function createMeshItem (id) {
 93 		var geometry	= getGeometry(id);
 94 		if( !geometry )	return null;
 95 		var mesh	= new THREE.Mesh( geometry, material );
 96 		return tQuery(mesh).geometry().scaleBy(1/16).back();
 97 	};	
 98 
 99 	// create the canvas element
100 	var canvas	= document.createElement('canvas');
101 	canvas.width	= 256;
102 	canvas.height	= 256;
103 	var context	= canvas.getContext('2d');
104 	var material	= getMaterial(canvas, true);
105 	// init some constants
106 	var tileUvW	= 1/canvas.width;
107 	var tileUvH	= 1/canvas.height;
108 	
109 
110 	var geometries	= [];
111 
112 	// load the image
113 	var items	= new Image();
114 	items.onload	= function(){
115 		// clear the canvas - TODO is this necessary ?
116 		context.clearRect(0, 0, canvas.width, canvas.height);
117 		// draw the loaded image to the canvas
118 		context.drawImage(items, 0, 0);
119 		// trigger the 'load' event
120 		this.trigger("load");
121 	}.bind(this);
122 	items.src = this._opts.url;
123 	
124 	// setup the public function
125 	this.createMesh		= createMeshItem;
126 	this.createGeometry	= getGeometry;
127 });
128

130 tQuery.MicroeventMixin(tQuery.MinecraftItems.prototype);
